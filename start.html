KHỞI TẠO TRẬN ĐẤU (Bản sửa lỗi đồng bộ P1 & P2) async function initBattle() { if (!user) { alert("Bạn cần đăng nhập!"); window.location.href='index.html'; return; } const urlParams = new URLSearchParams(window.location.search); const joinRoomId = urlParams.get('room'); if (joinRoomId) { // --- NGƯỜI CHƠI 2 --- roomId = joinRoomId; isP1 = false; // Bước 1: Lắng nghe trước để không bỏ lỡ tín hiệu listenRealtime(); // Bước 2: Cập nhật thông tin để "kích hoạt" trận đấu cho cả 2 const { data, error } = await supabaseClient.from('rooms').update({ p2_id: user.id, p2_name: user.email.split('@')[0], status: 'playing' }).eq('id', roomId).select(); if(error || !data || data.length === 0) { alert("Phòng không tồn tại hoặc đã đầy!"); window.location.href = 'start.html'; // Quay về nếu lỗi return; } // Bước 3: P2 chủ động vào game với dữ liệu vừa update showGameUI(data[0]); } else { // --- NGƯỜI CHƠI 1 --- isP1 = true; const { data, error } = await supabaseClient.from('rooms').insert([{ p1_id: user.id, p1_name: user.email.split('@')[0], status: 'waiting', p1_idx: 0, p2_idx: 0, p1_score: 0, p2_score: 0 }]).select(); if (data) { roomId = data[0].id; const fullLink = window.location.origin + window.location.pathname + '?room=' + roomId; document.getElementById('invite-link').innerText = fullLink; document.getElementById('room-link-container').classList.remove('hidden'); document.getElementById('lobby-status').innerText = "Đang đợi đối thủ tham gia..."; // P1 bắt đầu lắng nghe, khi P2 update status='playing', P1 sẽ tự vào game listenRealtime(); } } } // 2. LẮNG NGHE ĐỒNG BỘ (Sửa điều kiện lọc payload) function listenRealtime() { supabaseClient.channel(`room:${roomId}`) .on('postgres_changes', { event: 'UPDATE', schema: 'public', table: 'rooms', filter: `id=eq.${roomId}` }, payload => { const room = payload.new; // Cập nhật điểm số hiển thị liên tục cho cả 2 document.getElementById('score-p1').innerText = room.p1_score; document.getElementById('score-p2').innerText = room.p2_score; // Cập nhật điểm số vào biến local của mình playersData.p1_score = room.p1_score; playersData.p2_score = room.p2_score; // CHỈ chuyển sang màn hình chơi nếu đang ở sảnh (lobby) const isCurrentlyInLobby = !document.getElementById('screen-lobby').classList.contains('hidden'); if (room.status === 'playing' && isCurrentlyInLobby) { showGameUI(room); } // Kiểm tra kết thúc khi cả hai đã vượt qua số câu hỏi if (room.p1_idx >= myData.length && room.p2_idx >= myData.length) { showResult(); } }).subscribe(); }
